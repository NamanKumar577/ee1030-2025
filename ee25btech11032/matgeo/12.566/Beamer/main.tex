\begin{frame}{Question}
The product of the  eigenvalues of the matrix
\begin{align*}
    \myvec{2 & 3 \\ 0 & 7}
\end{align*} 
is 
\end{frame}


\begin{frame}{Theoretical Solution}
For a Sqaure Matrix $\vec{A}$

\begin{align}
    \mydet{\vec{A}} = \text{Product of Eigen Values}
\end{align}

\begin{align}
    \mydet{2 & 3 \\ 0 & 7} = 14 - 0  
\end{align}

Hence, Product of Eigen Values $= 14 $
\end{frame}
