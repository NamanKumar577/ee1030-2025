\subsection*{Question: } 
The eigenvalues of the matrix
\begin{align}
    \vec{P} = \myvec{4 & -5 \\ 2 & -5}
\end{align}
are 
\begin{enumerate}
    \begin{multicols}{2}
        \item $-7$ and $8$
        \item $-6$ and $5$
        \item $3$ and $4$
        \item $1$ and $2$
    \end{multicols}
\end{enumerate}

\textbf{Solution}:\\

\begin{align}
    \mydet{\vec{P} - \lambda\vec{I}} = 0
\end{align}
\begin{align}
    \mydet{\myvec{4-\lambda & -5 \\ 2 & -5-\lambda}} = 0 
\end{align}
\begin{align}
    \lambda^2 + \lambda -10 = 0
\end{align}
\begin{align}
    \lambda_1 = \frac{-1+\sqrt{41}}{2}, \quad 
    \lambda_2 = \frac{-1-\sqrt{41}}{2}
\end{align}

Hence, Answer : NO CORRECT OPTION
