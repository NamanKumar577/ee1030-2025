	\vec{A}\vec{A}^{-1}=\vec{I}

Using Augmented matrix,
\begin{align}
	\augvec{2}{2}{2&1&1&0\\7&4&0&1}
\end{align}
$R_2=R_2-3R_1$
\begin{align}
	\augvec{2}{2}{2&1&1&0\\1&1&-3&1}
\end{align}
$R_1=R_1-R_2$
\begin{align}
	\augvec{2}{2}{1&0&4&-1\\1&1&-3&1}
\end{align}
$R_2=R_2-R_1$
\begin{align}
	\augvec{2}{2}{1&0&4&-1\\0&1&-7&2}
\end{align}
\begin{align}
	\vec{A}^{-1}=\myvec{4&-1\\-7&2}
\end{align}
C function to find inverse:
\begin{lstlisting}
int inverse2x2(double mat[4], double inv[4]) {
    double a = mat[0], b = mat[1];
    double c = mat[2], d = mat[3];

    double det = a*d - b*c;
    if(det == 0.0) {
        return -1;  // not invertible
    }

    inv[0] =  d / det;
    inv[1] = -b / det;
    inv[2] = -c / det;
    inv[3] =  a / det;

    return 0; // success
}
\end{lstlisting}
